Terraform Provider

  cidr_block = "10.0.0.0/16"
}

resource "mockfactory_lambda_function" "api" {
  function_name = "my-api"
  runtime       = "python3.9"
  memory_size   = 256
}

Same code works for real AWS - just swap the provider!

Ansible Module

- name: Create VPC
  mockfactory:
    resource_type: vpc
    action: create
    params:
      cidr_block: "10.0.0.0/16"

- name: Create Lambda
  mockfactory:
    resource_type: lambda
    action: create
    params:
      function_name: "my-api"
